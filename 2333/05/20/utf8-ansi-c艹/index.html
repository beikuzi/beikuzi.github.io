void getfiles(string path)
{
long jubing=0;
struct _finddata_t fi;//fi fileinfo
string p;
if((jubing = _findfirst(p.assign(path).append(“\*”).c_str(), &fi))!=-1)
{//这种相加和直接string加起来一样是不赋值的
do
{
if(fi.attrib & _A_SUBDIR)
{//是个目录
if(strcmp(fi.name,”.”)!=0 and strcmp(fi.name,”..”)!=0)
{//不是当前目录或者父目录
getfiles(p.assign(path).append(“\“).append(fi.name));
}//就是子目录咯
}
else//是个普通的文件呢
deal(fi.name);
}while(_findnext(jubing,&fi)==0);//能找到下一个
_findclose(jubing);
}
}
