void pic_size(Mat pic_in,Mat pic_out,Size x)

{ 

resize(pic_in, pic_out, x, 0, 0, INTER_LINEAR);  

//x是size的标准形式 Size(x,y) //Size(1280,720)  

//两个0是缩放系数的值dx、dy，默认为0时就是现在宽度/原列数  

/*  

INTER_NEAREST - 最近邻插值  

INTER_LINEAR - 线性插值（默认）  

INTER_AREA - 区域插值  

INTER_CUBIC - 三次样条插  

INTER_LANCZOS4 - Lanczos插值  

*/

}

void vid_read(string name,VideoCapture cap)

{

    imshow(name, vid);  

}

}

//10.录像一定时间的视频

void vid_settime_cap(string name,VideoCapture& cap, VideoWriter& wcap,int time,int fps)

{ 

Mat vid;  

int k=0;  

while (1)  

{  

    cap >> vid;  

    wcap << vid;  

    if (waitKey(30) >= 0)break;  

    imshow(name, vid);  

    k++; if (k >= time * fps)break;  

}

}
